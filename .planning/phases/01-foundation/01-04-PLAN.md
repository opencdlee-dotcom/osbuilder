    ```python
    def parse_state_md(text: str) -> dict:
        """Parse `key: value` lines from state.md. Ignores blank lines and `#` headers."""
        fields: dict = {}
        for line in text.splitlines():
            line = line.strip()
            if not line or line.startswith("#"):
                continue
            key, sep, value = line.partition(":")
            if sep:
                fields[key.strip()] = value.strip()
        return fields
    ```

    **C. `atomic_write(path: Path, content: str) -> None`** тАФ paste from RESEARCH.md Pattern 3 + Code Examples lines 767-781:

    ```python
    def atomic_write(path: Path, content: str) -> None:
        """Write content to path atomically: tmp file in same dir тЖТ os.replace.

        Atomic on POSIX + Windows since Python 3.3 (per os.replace docs).
        Same-directory tmp file is required (atomic-rename only works within one FS).
        """
        path.parent.mkdir(parents=True, exist_ok=True)
        fd, tmp_path = tempfile.mkstemp(
            dir=str(path.parent), prefix=".state.", suffix=".tmp"
        )
        try:
            with os.fdopen(fd, "w", encoding="utf-8", newline="\n") as f:
                f.write(content)
                f.flush()
                os.fsync(f.fileno())
            os.replace(tmp_path, path)  # NOT os.rename тАФ Pitfall 8
        except Exception:
            try:
                os.unlink(tmp_path)
            except FileNotFoundError:
                pass
            raise
    ```

    ```python
    _UNSAFE_VALUE_PATTERN = re.compile(r"[\n\r]|\.\.")
    # Rejects:
    #   \n, \r           (T-1-V5: newline injection corrupts key:value-per-line format)
    #   ..               (T-1-V12: path-traversal in value content)


    def _resolve_project_root(arg: str | None) -> Path:
        """Resolve --project-root to an absolute, existing directory.

        If arg is None, walk up from cwd looking for .planning/.
        Exit non-zero with friendly message if not found.
        """
        if arg is not None:
            root = Path(arg).resolve()
            if not root.is_dir():
                print(f"[state_writer] error: --project-root '{arg}' is not a directory.",
                      file=sys.stderr)
                sys.exit(2)
            return root
        # Walk up looking for .planning/
        cwd = Path.cwd().resolve()
        for parent in [cwd, *cwd.parents]:
            if (parent / ".planning").is_dir():
                return parent
        print(
            "[state_writer] error: no .planning/ found above cwd. "
            "Pass --project-root <abs-path> or run from inside a project.",
            file=sys.stderr,
        )
        sys.exit(2)


    def _state_md_path(project_root: Path) -> Path:
        return project_root / ".planning" / "osbuilder" / "state.md"


    def _check_field_allowed(field: str, *, allow_for_bump: bool = False) -> None:
        """Cheap allowlist check тАФ should run BEFORE _check_value_safe (SUGGESTION 2)."""
        if field not in REQUIRED_FIELDS:
            print(
                f"[state_writer] error: unknown --field '{field}'. "
                f"Allowed: {', '.join(REQUIRED_FIELDS)}.",
                file=sys.stderr,
            )
            sys.exit(2)
        if allow_for_bump and field != "retry_count":
            print(
                f"[state_writer] error: bump/reset only valid for retry_count, not '{field}'.",
                file=sys.stderr,
            )
            sys.exit(2)


    def _check_value_safe(value: str) -> None:
        """T-1-V5 + T-1-V12: reject newline/CR and `..` in --value.

        Order: this MUST be called AFTER _check_field_allowed (SUGGESTION 2 тАФ defense
        in depth: fail fast on the cheap allowlist check before doing regex scan).
        """
        if _UNSAFE_VALUE_PATTERN.search(value):
            print(
                "[state_writer] error: --value rejected. "
                "Newlines, carriage returns, and `..` are not allowed in field values.",
                file=sys.stderr,
            )

        root = _resolve_project_root(args.project_root)
        path = _state_md_path(root)

                f"Missing fields: {', '.join(missing)}. "
                f"Run `state_writer.py init --goal \"<recover>\"` to reset.",
                file=sys.stderr,
            )
            return 2
        if args.field:
            _check_field_allowed(args.field)
            print(fields.get(args.field, ""))
            return 0
        if args.format == "json":
            print(json.dumps(fields))
        else:
            print(text, end="")
        return 0


    def _cmd_bump(args: argparse.Namespace) -> int:
        root = _resolve_project_root(args.project_root)
        path = _state_md_path(root)
        # SUGGESTION 2: field-allowed first (no --value here, so order is just for parity)
        _check_field_allowed(args.field, allow_for_bump=True)
        fields = parse_state_md(path.read_text(encoding="utf-8"))

        except ValueError:

        if current >= 3:
            print(f"[state_writer] error: {args.field} already at cap (3).",
                  file=sys.stderr)
            return 2
        fields[args.field] = str(current + 1)
        atomic_write(path, render_state_md(fields))
        return 0


    def _cmd_reset(args: argparse.Namespace) -> int:

        path = _state_md_path(root)
        _check_field_allowed(args.field, allow_for_bump=True)
        fields = parse_state_md(path.read_text(encoding="utf-8"))
        fields[args.field] = "0"
        atomic_write(path, render_state_md(fields))
        return 0


    def _cmd_validate(args: argparse.Namespace) -> int:
        root = _resolve_project_root(args.project_root)
        path = _state_md_path(root)
        if not path.exists():
            print(f"[state_writer] error: state.md not found at {path}",
                  file=sys.stderr)
            return 2
        fields = parse_state_md(path.read_text(encoding="utf-8"))
        missing = validate(fields)
        if missing:
            print(f"[state_writer] error: state.md missing fields: {', '.join(missing)}",
                  file=sys.stderr)
            return 2
        print(f"[state_writer] state.md at {path} is valid.")
        return 0


    def main(argv: list[str] | None = None) -> int:
        parser = argparse.ArgumentParser(prog="state_writer.py", description=__doc__)
        parser.add_argument("--project-root", default=None,
                            help="Absolute path to project root (default: walk up for .planning/)")
        sub = parser.add_subparsers(dest="cmd", required=True)

        p_init = sub.add_parser("init")
    ```
