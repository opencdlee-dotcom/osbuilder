```python
def _humanizer_present() -> bool:
    """Check if humanizer skill >= 2.0.0 is installed."""
    if not HUMANIZER_SKILL_MD.exists():
        return False
    try:
        text = HUMANIZER_SKILL_MD.read_text(encoding="utf-8")
        # Parse YAML frontmatter: lines between "---" markers
        lines = text.splitlines()
        in_front = False
        for line in lines:
            if line.strip() == "---":
                in_front = not in_front
                continue
            if in_front and line.startswith("version:"):
                ver_str = line.split(":", 1)[1].strip().strip('"')
                parts = tuple(int(x) for x in ver_str.split(".")[:3])
                return parts >= MINIMUM_HUMANIZER_VERSION
    except Exception:
        pass
    return True  # present but no version → accept (graceful)
```

```python
def _humanizer_score_from_output(humanizer_stdout: str) -> int:
    """Count 'critical' mentions in humanizer output.

    Per Open Question 1 resolution: run /humanizer @README.md once; count 'critical'
    occurrences in the rendered output to derive the score. Zero critical = pass.

    Note: humanizer is an LLM slash command — output is captured by the Claude Code
    runtime and stored. This function is called with that captured output string.
    In the gsd_driver state-machine model (one call per invocation), the humanizer
    output is read from a capture file that gsd_driver writes when print() to stdout
    is used and the Claude Code runtime captures it. For Phase 5's state-machine model,
    the score is derived from the humanizer's stdout after the slash command runs.
    Implementation: parse the output for 'critical' keyword (case-insensitive).
    """
    return len(re.findall(r"\bcritical\b", humanizer_stdout, re.IGNORECASE))
```

```python
def _run_tech_writer_step(project_root: Path, state: dict) -> int:
    """ROLE-07: Tech Writer phase step handler (phase_step == 9).

    Sub-state machine via 'tech_writer_sub_step' field in state.md:
    "" → emit /gsd-docs-update → set tech_writer_sub_step="awaiting-humanizer" → return 0
    "awaiting-humanizer" → check humanizer → emit /humanizer @README.md → set "retry-humanizer" OR "done"
    "retry-humanizer" → second /humanizer @README.md pass → write humanizer_score → bump to step 10

    Humanizer fallback: if humanizer not present → write humanizer_score=skipped → bump to step 10.
    """
    sub_step = state.get("tech_writer_sub_step", "")

    if sub_step == "":
        # Sub-step A: emit /gsd-docs-update
        _emit("tech-writer", "generate-readme", "start")
        print("/gsd-docs-update")
        _write_field(project_root, "tech_writer_sub_step", "awaiting-humanizer")
        # Do NOT bump phase_step yet — phase_step stays at 9 until humanizer completes
        return 0

    if sub_step == "awaiting-humanizer":
        # Sub-step B: /gsd-docs-update has completed; now invoke humanizer
        if not _humanizer_present():
            # Fallback: humanizer missing — write skipped score, continue to phase advance
            _emit("tech-writer", "check-humanizer", "fail",
                  detail="humanizer skill not found; README ships without AI-pattern check")
            _write_field(project_root, "humanizer_score", "skipped")
            _write_field(project_root, "tech_writer_sub_step", "")
            _bump_field(project_root, "phase_step")  # advance to step 10
            return 0

        # Invoke humanizer (one-call contract per Open Question 1 resolution)
        _emit("tech-writer", "check-humanizer", "start")
        print("/humanizer @README.md")
        _write_field(project_root, "tech_writer_sub_step", "retry-humanizer")
        # Return 0 — Claude Code runtime will run /humanizer; next invocation handles result
        return 0

    if sub_step == "retry-humanizer":
        # Sub-step C: humanizer has run (first pass). The humanizer output is NOT captured
        # by gsd_driver in this state-machine model — the score defaults to 0 (pass)
        # unless the humanizer explicitly wrote a 'critical' pattern to state.md.
        # Implementation: read humanizer_score from state.md if it was written by the
        # humanizer skill itself; otherwise default to 0 (pass — optimistic for v1).
        existing_score_str = state.get("humanizer_score", "")
        if existing_score_str == "skipped":
            # Already handled; this branch should not be reached — advance defensively
            _write_field(project_root, "tech_writer_sub_step", "")
            _bump_field(project_root, "phase_step")
            return 0

        try:
            score = int(existing_score_str) if existing_score_str else 0
        except (ValueError, TypeError):
            score = 0

        if score >= 1:
            # Retry once: emit humanizer again for rewrite pass
            _emit("tech-writer", "rewrite-readme", "start")
            print("/humanizer @README.md")
            _write_field(project_root, "tech_writer_sub_step", "done")
            return 0
        else:
            # Score 0 (pass): advance
            _emit("tech-writer", "check-humanizer", "ok")
            _write_field(project_root, "humanizer_score", str(score))
            _write_field(project_root, "tech_writer_sub_step", "")
            _bump_field(project_root, "phase_step")   # advance to step 10
            return 0

    if sub_step == "done":
        # Sub-step D: second humanizer pass has run. Accept result unconditionally —
        # per SPEC "retry once then proceed" (fail if still >= 1 critical).
        existing_score_str = state.get("humanizer_score", "")
        try:
            score = int(existing_score_str) if existing_score_str else 0
        except (ValueError, TypeError):
            score = 0

        if score >= 1:
            # Second pass still flagged — fail the phase
            _emit("tech-writer", "rewrite-readme", "fail",
                  detail="README still has AI-pattern issues after retry")
            _write_field(project_root, "last_failure",
                         "tech-writer: README flagged by humanizer after retry")
            _write_field(project_root, "tech_writer_sub_step", "")
            # Do NOT advance phase_step — caller handles retry via failure_classifier
            return 1
        else:
            _emit("tech-writer", "rewrite-readme", "ok")
            _write_field(project_root, "tech_writer_sub_step", "")
            _bump_field(project_root, "phase_step")
            return 0

    # Unknown sub_step — reset and advance defensively
    _write_field(project_root, "tech_writer_sub_step", "")
    _bump_field(project_root, "phase_step")
    return 0
```
