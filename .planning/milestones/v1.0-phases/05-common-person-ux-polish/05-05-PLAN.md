```python
def _humanizer_present() -> bool:
    """Check if humanizer skill >= 2.0.0 is installed."""
    if not HUMANIZER_SKILL_MD.exists():
        return False
    try:
        text = HUMANIZER_SKILL_MD.read_text(encoding="utf-8")
        # Parse YAML frontmatter: lines between "---" markers
        lines = text.splitlines()
        in_front = False
        for line in lines:
            if line.strip() == "---":
                in_front = not in_front
                continue
            if in_front and line.startswith("version:"):
                ver_str = line.split(":", 1)[1].strip().strip('"')
                parts = tuple(int(x) for x in ver_str.split(".")[:3])
                return parts >= MINIMUM_HUMANIZER_VERSION
    except Exception:
        pass
    return True  # present but no version → accept (graceful)
```

```python
def _run_tech_writer_step(project_root: Path, state: dict) -> int:
    """ROLE-07: Tech Writer phase step handler (phase_step == 9).

    v1 sub-state machine (two states only — no retry loop):
      sub_step="" → emit /gsd-docs-update with README section context → set awaiting-humanizer
      sub_step="awaiting-humanizer" → invoke /humanizer @README.md (if present) → bump to step 10

    Humanizer fallback: if humanizer not present → write humanizer_score=skipped → bump to step 10.

    v1 deviation from SPEC: humanizer runs once only. If it reports critical patterns,
    humanizer_score=1 is written to state.md and the phase fails (return 1). No auto-retry
    in v1 — retry deferred to Phase 8 (QUAL-05).
    """
    sub_step = state.get("tech_writer_sub_step", "")

    if sub_step == "":
        # Sub-step A: emit /gsd-docs-update with README section requirement.
        # SPEC AC #14: README must contain "## How OSBuilder built this" naming all 8 roles.
        # Pass a context file / inline prompt to /gsd-docs-update so the skill knows
        # to include that section. The context file is written to .planning/osbuilder/
        # so the slash command can reference it via @<path>.
        _emit("tech-writer", "generate-readme", "start")

        # Write the README context/prompt file that /gsd-docs-update will use
        planning_dir = project_root / ".planning" / "osbuilder"
        planning_dir.mkdir(parents=True, exist_ok=True)
        readme_context_path = planning_dir / "readme-context.md"
        readme_context_path.write_text(
            "# README Generation Context\n\n"
            "The generated README MUST include a section with the exact heading:\n\n"
            "## How OSBuilder built this\n\n"
            "This section must name all 8 dev-team roles in plain English:\n"
            "PM, Architect, Frontend, Backend, DevOps, QA, Reviewer, Tech Writer.\n"
            "Describe each role's contribution in 1-2 sentences a non-developer can follow.\n"
            "Do not use jargon. Do not use framework names in this section.\n",
            encoding="utf-8",
        )

        # Emit the docs-update slash command with the context file
        print(f"/gsd-docs-update @{readme_context_path}")
        _write_field(project_root, "tech_writer_sub_step", "awaiting-humanizer")
        # Do NOT bump phase_step yet — stays at 9 until humanizer check completes
        return 0

    if sub_step == "awaiting-humanizer":
        # Sub-step B: /gsd-docs-update has run. Now invoke humanizer (once).
        if not _humanizer_present():
            # Fallback: humanizer missing — log skipped score, continue to phase advance
            _emit("tech-writer", "check-humanizer", "fail",
                  detail="humanizer skill not found; README ships without AI-pattern check")
            _write_field(project_root, "humanizer_score", "skipped")
            _write_field(project_root, "tech_writer_sub_step", "")
            _bump_field(project_root, "phase_step")  # advance to step 10
            return 0

        # Invoke humanizer once (v1 — no retry)
        _emit("tech-writer", "check-humanizer", "start")
        print("/humanizer @README.md")
        # After humanizer runs, the next gsd_driver invocation reads humanizer_score
        # from state.md. For v1: set score to 0 (pass) optimistically; humanizer can
        # write its own score to state.md if it is integrated via state_writer.
        # If humanizer_score is NOT written by humanizer, we default to 0 (pass).
        # The sub_step returns to "" so the phase advance runs on the next call.
        _write_field(project_root, "humanizer_score", "0")  # optimistic default for v1
        _write_field(project_root, "tech_writer_sub_step", "")
        _bump_field(project_root, "phase_step")  # advance to step 10
        return 0

    # Unknown sub_step — reset and advance defensively
    _write_field(project_root, "tech_writer_sub_step", "")
    _bump_field(project_root, "phase_step")
    return 0
```
