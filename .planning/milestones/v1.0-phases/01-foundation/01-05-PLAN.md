    ```powershell
    # OSBuilder bootstrap shim — Windows
    # Source: .planning/phases/01-foundation/01-RESEARCH.md Pattern 6 (lines 552-599)
    # Purpose: ensure Python ≥3.10 is installed on Windows via winget,
    # then (if Python was already present) re-exec into state_writer.py.
    # PowerShell 5.1 compatible (the version built into Windows 10/11).
    # For macOS / Linux, use bootstrap.sh.

    $ErrorActionPreference = "Stop"

    $SkillDir = Join-Path $HOME ".claude\skills\osbuilder"
    $StateWriter = Join-Path $SkillDir "scripts\state_writer.py"

    # --- 0. Track whether Python was already on PATH BEFORE we did anything (BLOCKER 3) ---
    # If Python was already present, we can `& python ...` after the script completes.
    # If we just installed Python via winget, the current shell does NOT see the new PATH
    # (Pitfall 5) — we must exit 0 with a "reopen shell" message instead.
    $pythonAlreadyPresent = $null -ne (Get-Command python -ErrorAction SilentlyContinue)
    $JustInstalled = $false

    # --- 1. Already have Python ≥ 3.10? -----------------------------------------
    $needInstall = $true
    if ($pythonAlreadyPresent) {
        $verLine = & python --version 2>&1

        # Pitfall 6: detect Microsoft Store python.exe alias.
        # The alias's --version output is empty / non-matching.
        if ($verLine -notmatch "Python (\d+)\.(\d+)") {
            Write-Warning "[bootstrap] 'python' is the Microsoft Store alias, not a real Python install."
            Write-Warning "[bootstrap] To fix: Settings -> Apps -> Advanced app settings -> App execution aliases"
            Write-Warning "[bootstrap] then turn OFF python.exe and python3.exe, then re-run this script."
            # Treat the alias-only state as "Python not present" for re-exec purposes.
            $pythonAlreadyPresent = $false
            # Continue to install — winget install will work; alias still active until disabled.
        } else {
            $major = [int]$Matches[1]
            $minor = [int]$Matches[2]
            if ($major -ge 3 -and $minor -ge 10) {
                Write-Host "[bootstrap] Python $major.$minor detected -- OK."
                $needInstall = $false
            } else {
                Write-Host "[bootstrap] Python $major.$minor found but >= 3.10 required."
            }
        }
    }

    if ($needInstall) {
        # --- 2. Detect winget (built into Windows 10+/11) ---------------------
        $winget = Get-Command winget -ErrorAction SilentlyContinue
        if ($null -eq $winget) {
            Write-Error "[bootstrap] winget not found. Update Windows or install winget from the Microsoft Store."
            exit 1
        }

        # --- 3. Install Python 3.13 via winget --------------------------------
        Write-Host "[bootstrap] Installing Python 3.13 via winget..."
        & winget install -e --id Python.Python.3.13 --silent --accept-source-agreements --accept-package-agreements
        if ($LASTEXITCODE -ne 0) {
            Write-Error "[bootstrap] winget install failed (exit $LASTEXITCODE)."
            exit 1
        }
        $JustInstalled = $true

        # --- 4. Verify post-install -------------------------------------------
        # Pitfall 5: shell PATH does not auto-refresh after winget install.
        $verify = Get-Command python -ErrorAction SilentlyContinue
        if ($null -eq $verify) {
            Write-Warning "[bootstrap] Python installed, but the current shell's PATH does not see it yet."
        } else {
            $verLine = & python --version 2>&1
            if ($verLine -match "Python (\d+)\.(\d+)") {
                Write-Host "[bootstrap] Done. $verLine"
            } else {
                # Likely the Microsoft Store alias is still intercepting.
                Write-Warning "[bootstrap] Python installed via winget, but 'python' is still the MS Store alias."
                Write-Warning "[bootstrap] Disable the alias: Settings -> Apps -> Advanced app settings -> App execution aliases"
                Write-Warning "[bootstrap] then turn OFF python.exe and python3.exe."
            }
        }
    }

    # --- 5. Two-mode re-exec into state_writer.py (BLOCKER 3) -------------------
    # Mode A: Python was ALREADY on PATH at script start ($pythonAlreadyPresent = $true).
    #         The current shell has python on PATH — we can invoke state_writer.py.
    # Mode B: Python was just installed by winget in this run ($JustInstalled = $true).
    #         The current shell does NOT see the new PATH (Pitfall 5).
    #         Exit 0 with "reopen shell" message — user re-runs from a fresh shell,
    #         which will then take Mode A.

    if ($JustInstalled) {
        Write-Host ""
        Write-Host "[bootstrap] Python install complete. PowerShell does not refresh PATH automatically."
        Write-Host "[bootstrap] Close this window, open a new PowerShell, and re-run /osbuilder to continue."
        exit 0
    }

    if ($pythonAlreadyPresent) {
        if (Test-Path $StateWriter) {
            Write-Host "[bootstrap] Invoking state_writer.py..."
            & python $StateWriter @args
            exit $LASTEXITCODE
        } else {
            Write-Host "[bootstrap] Bootstrap complete. Run /osbuilder to continue."
            # state_writer.py is copied into $SkillDir\scripts\ by install.sh (Plan 03).
            # On a partial-state install, this branch fires.
            exit 0
        }
    }

    # Fallback: Python was alias-only, we installed via winget, but Get-Command still
    # returns null. Exit 0 — user remediation already documented above.
    exit 0
    ```
