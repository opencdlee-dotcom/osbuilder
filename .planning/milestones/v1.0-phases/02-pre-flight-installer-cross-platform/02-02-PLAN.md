INTERFACE 6 — Atomic write (carry-forward from `state_writer.py:100-112`). EXTRACT or DUPLICATE; do NOT `from state_writer import atomic_write` (D-05: scripts/ files are independent — relative imports forbidden):

```python
def atomic_write(path: Path, content: str) -> None:
    """Atomic file write via os.replace (atomic on POSIX + Windows)."""
    path.parent.mkdir(parents=True, exist_ok=True)
    tmp = path.with_name(f".{path.stem}.{os.getpid()}.tmp")
    try:
        tmp.write_text(content, encoding="utf-8")
        os.replace(str(tmp), str(path))
    except BaseException:
        try:
            tmp.unlink(missing_ok=True)
        except Exception:
            pass
        raise
```

INTERFACE 8 — Linux distro detection (from RESEARCH.md lines 982-995):

```python
def detect_linux_manager() -> str | None:
    """Returns 'apt-get', 'dnf', or None if unsupported."""
    try:
        osr = platform.freedesktop_os_release()
    except (OSError, NotImplementedError):
        return None
    ids = {osr.get("ID", ""), *osr.get("ID_LIKE", "").split()}
    if "debian" in ids or "ubuntu" in ids:
        return "apt-get"
    if "fedora" in ids or "rhel" in ids:
        return "dnf"
    return None
```

```python
def _confirm_batch(plan: Plan) -> bool:
    """Single y/n covering the whole batch. Returns True if user typed y/yes (case-insensitive); False otherwise."""
    if not sys.stdin.isatty():
```

INTERFACE 11 — apply() flow with auto-rollback (RESEARCH.md lines 184-202):

```python
def apply(plan: Plan) -> int:
    """Execute plan; auto-rollback on any failure."""
    log = _read_install_log()  # may be empty
    for action in plan.actions:
        # Step 1: record "started" BEFORE subprocess.run (atomicity invariant; T-02 mitigation)
        entry = _new_log_entry(action, status="started")
        log["actions"].append(entry)
        _write_install_log(log)  # atomic_write

        # Step 2: run install
        try:
            # On sudo-required calls: do NOT capture_output — let sudo own the TTY (Pitfall 5)
            cap = not action.requires_sudo
            result = subprocess.run(action.install_argv, shell=False,
                                    capture_output=cap, text=cap, check=False)
        except KeyboardInterrupt:
            entry["status"] = "failed"
            _write_install_log(log)
            return rollback()  # roll back everything in REVERSE
        # Step 3: success path — mark succeeded
        if result.returncode == 0:
            entry["status"] = "succeeded"
            entry["succeeded_at"] = _now_iso()
            _write_install_log(log)
        else:
            entry["status"] = "failed"
            _write_install_log(log)
            return rollback()
    return 0
```

NB: argv is ALWAYS list-form (`subprocess.run([...], shell=False)`). NEVER construct shell strings. Pitfall 14 mitigation.

       ```python
       def _probe_version(tool: str) -> str | None:
           """Run `<tool> --version` and return the trimmed stdout, or None on failure."""
           which = shutil.which(tool)
           if which is None:
               return None
           try:
               r = subprocess.run([tool, "--version"], capture_output=True, text=True,
                                  shell=False, timeout=5)
           except (subprocess.TimeoutExpired, FileNotFoundError, OSError):
               return None
           return (r.stdout or r.stderr or "").strip() or None

       def _node_version_ok(version_str: str | None) -> bool:
           """`v20.10.0` -> 20 >= 20 -> True."""
           if not version_str:
               return False
           try:
               major = int(version_str.lstrip("v").split(".")[0])
           except (ValueError, IndexError):
               return False
           return major >= NODE_MIN_MAJOR

       def _python_version_ok(version_str: str | None) -> bool:
           """`Python 3.13.0` -> (3, 13) >= (3, 13) -> True."""
           if not version_str:
               return False
           try:
               parts = version_str.replace("Python", "").strip().split(".")
               major, minor = int(parts[0]), int(parts[1])
           except (ValueError, IndexError):
               return False
           return (major, minor) >= PYTHON_MIN
       ```

       ```python
               return 0
           # Print preview + single confirmation prompt (PRE-02 + PRE-05)
           sys.stdout.write(render_preview(plan))
           if not _confirm_batch(plan):

               return 1
           log = _read_install_log()
           for action in plan.actions:
               # ATOMICITY INVARIANT (T-02-11 high-severity mitigation):
               # write 'started' to log BEFORE subprocess.run is invoked.
               # The strict ordering write→subprocess is the falsifying claim
               # exercised by test_log_recorded_before_subprocess.
               entry = _new_log_entry(action, platform_name=plan.os)
               log["actions"].append(entry)
               _write_install_log(log)
               # On sudo-required calls, do NOT capture stdin/stdout — let sudo own
               # the TTY (Pitfall 5).
               cap = not action.requires_sudo
               try:
                   result = subprocess.run(
                       action.install_argv,
                       shell=False,
                       capture_output=cap,
                       text=cap,
                       check=False,
                   )
               except KeyboardInterrupt:
                   entry["status"] = "failed"
                   _write_install_log(log)
                   sys.stderr.write("\nInterrupted; rolling back...\n")
                   return rollback()
               except (FileNotFoundError, OSError) as e:
                   entry["status"] = "failed"
                   _write_install_log(log)
                   sys.stderr.write(f"OSBuilder: {action.tool} install failed: {e}\n")
                   return rollback()
               # Re-write log to reflect success or failure
               if result.returncode == 0:
                   entry["status"] = "succeeded"
                   entry["succeeded_at"] = _now_iso()
                   _write_install_log(log)
               else:
                   entry["status"] = "failed"
                   _write_install_log(log)
                   sys.stderr.write(
                       f"OSBuilder: {action.tool} install exited {result.returncode}; "
                       f"rolling back...\n"
                   )
                   return rollback()
           return 0
       ```

       ```python
       def rollback() -> int:
           """Iterate install-log in REVERSE; run each succeeded action's uninstall_argv."""
           log = _read_install_log()
           # Walk in reverse over actions where status == "succeeded" or "started"
           # (started but not succeeded means a partial install; attempt cleanup anyway)
           failed = 0
           for entry in reversed(log["actions"]):
               if entry["status"] in ("rolled-back",):
                   continue
               try:
                   r = subprocess.run(entry["uninstall_argv"], shell=False,
                                      capture_output=False, check=False)
               except (FileNotFoundError, OSError):
                   failed += 1
                   continue
               if r.returncode == 0:
                   entry["status"] = "rolled-back"
               else:
                   failed += 1
           _write_install_log(log)
           return 0 if failed == 0 else 2
       ```
